Python Kodu:
import os
from astropy.io import fits
import numpy as np
import matplotlib.pyplot as plt

# Veri klasörü yolu
DATA_DIR = r"C:\Users\omerbyrm\Desktop\asteroid_project\data"

def read_and_display_fits(data_dir):
    """FITS dosyasını okur ve görüntüler"""
    # FITS dosyalarını bul
    fits_files = [f for f in os.listdir(data_dir) if f.endswith('.fits')]

    if not fits_files:
        print("HATA: FITS dosyası bulunamadı.")
        return

    # İlk dosyayı aç
    first_file = fits_files[0]
    file_path = os.path.join(data_dir, first_file)

    with fits.open(file_path) as hdul:
        image_data = hdul[0].data
        header = hdul[0].header

        # Görüntüyü çiz
        plt.figure(figsize=(10, 8))
        plt.imshow(image_data, cmap='gray', origin='lower')
        plt.title(f"FITS Görüntüsü: {first_file}")
        plt.show()
